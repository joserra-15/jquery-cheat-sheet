  e.preventDefault();
  document.querySelectorAll('#classAndDisplayNoneVanilla .border-and-Background')
    .forEach(element=>{
      if(window.getComputedStyle(element).getPropertyValue('display') === "none"){
        element.style.color = 'blue';

      }
    })

})
